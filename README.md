DT.Util.LOO = function(points, classes, classFunc, hLims = 0) {
  n = dim(points)[1]# –±–µ—Ä—ë–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –∏–∑ –ø–µ—Ä–≤–æ–≥–æ —Å—Ç–æ–ª–±—Ü–∞
  if(identical(classFunc, DT.kNN.kNN) || identical(classFunc, DT.WkNN.WkNN)) { #—Å—Ä–∞–≤–Ω–µ–Ω–∏–µ –º–µ—Ç–æ–¥–æ–≤,—Å–º –≤—ã—à–µ
  loo = double(n-1) #n-1 –ø–æ—Å–∫–æ–ª—å–∫—É –º—ã –≤—Å–µ–≥–¥–∞ —É–¥–∞–ª—è–µ–º –æ–¥–∏–Ω —ç–ª–µ–º–µ–Ω—Ç –¥–ª—è LOO
  
  for (i in 1:n) { #–∏–¥—ë–º –ø–æ –≤—ã–±–æ—Ä–∫–µ
  #–Ω–∞—Ö–æ–¥–∏–º —ç–≤–∫–ª–∏–¥–æ–≤—ã —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –æ—Ç –Ω–∞—à–µ–π —Ç–æ—á–∫–∏ –¥–æ –æ—Å—Ç–∞–≤—à–µ–π—Å—è –≤—ã–±–æ—Ä–∫–∏ –±–µ–∑ –Ω–∞—à–µ–π —Ç–æ—á–∫–∏ - points[-i]
    distances = DT.Util.getDist(points[-i,], points[i,], DT.Util.euclidDist) 
    names(distances) = classes[-i] #–ø—Ä–∏—Å–≤–∞–∏–≤–∞–µ–º –∏–º–µ–Ω–∞ –∫–ª–∞—Å—Å–æ–≤ —Ç–æ—á–µ–∫ –≤ –º–∞—Å—Å–∏–≤–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
    sortDist = sort(distances) #—Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—é
    #–≤–ª–æ–∂–µ–Ω–Ω—ã–π —Ü–∏–∫–ª –Ω—É–∂–µ–Ω –¥–ª—è –ø–æ–¥—Å—á—ë—Ç–∞ 
    for (l in 1:n - 1) { 
      bestClass = classFunc(sortDist, l) #–≤—ã–∑–æ–≤ kNN –¥–ª—è l —Å–æ—Å–µ–¥–µ–π
      loo[l] = loo[l] + ifelse(bestClass == classes[i], 0, 1) 
      } #–µ—Å–ª–∏ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ç–æ—Ä –æ—à–∏–±—Å—è, —É–≤–µ–ª–∏—á–∏–≤–∞–µ–º LOO –ø—Ä–∏ l —Å–æ—Å–µ–¥—è—Ö
    }
  } # –≤–æ—Ç —Ç–∞–∫ –≤–æ—Ç –ø—Ä–æ—Å—Ç–æ –º–æ–∂–Ω–æ –ø–æ–¥–µ–ª–∏—Ç—å –≤–µ—Å—å –º–∞—Å—Å–∏–≤ –Ω–∞ —á–∏—Å–ª–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤—ã–±–æ—Ä–∫–∏
    loo = loo / n
  return(loo)
}
